    <script>
    function enviarFormulario() {
        var nome = document.getElementById("nome").value;
        var cpf = document.getElementById("cpf").value;
        var matricula = document.getElementById("matricula").value;
        var codigoSecretaria = document.getElementById("codigoSecretaria").value;
        var viaCartao = document.getElementById("viaCartao").value;

        var url = "https://script.google.com/a/macros/recife.pe.gov.br/s/AKfycbw3MA7I36GNVkI2fmhEXf_IM5gjNNwjFHz0YxO9ISXRL-3rjidN5GQjTeNrLla5MQ_Ftg/exec"; // Substitua pelo URL do seu aplicativo da web

        fetch(url, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/x-www-form-urlencoded',
            },
            body: `nome=${nome}&cpf=${cpf}&matricula=${matricula}&codigoSecretaria=${codigoSecretaria}&viaCartao=${viaCartao}`,
        })
            .then((response) => response.text())
            .then((data) => {
                alert(data);
                document.getElementById('meuFormulario').reset();
            })
            .catch((error) => {
                console.error('Erro:', error);
            });
    }
    </script>
